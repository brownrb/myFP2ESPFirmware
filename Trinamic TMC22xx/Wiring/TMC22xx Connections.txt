/------------------------------------------------
/ METHOD1: CONNECTION FOR TMC2225 WITH DRV8825HW203-rev11 PCB USING J12-BT CONNECTOR
/------------------------------------------------
D10			J12-BTTXD	TMC2225-UARTpin
D11	1K R11 		J12-BTRXD	TMC2225-UARTpin	

#define BTTX                11
#define BTRX                10            

Wire VIO on underside of PCB to 5V.
Wire CLK to GND
